# 题目
给定一个非空字符串 s，最多删除一个字符。判断是否能成为回文字符串。


* 示例:

>示例 1:
输入: "aba"<br>
输出: True

>示例 2:
输入: "abca"<br>
输出: True<br>
解释: 你可以删除c字符。



* 代码1,用时720ms：
    ```C++
    class Solution {
        public:
            bool validPalindrome(string s) {
                int beg = 0,end = s.size()-1;
                while(end>beg)
                {
                    if(s[beg] != s[end])
                    {
                        if(s[beg+1] == s[end])
                        {
                            int newbeg=beg+2;
                            int newend=end-1;
                            bool flag = true;
                            while(newend>newbeg)
                            {
                                if(s[newbeg]!=s[newend])
                                {
                                    flag = false;
                                    break;
                                }
                                ++newbeg;
                                --newend;
                            }
                            if(flag)
                            {
                                return true;
                            }
                        }
                        else if(s[beg] == s[end-1])
                        {
                            int newend=end-2;
                            int newbeg = beg+1;
                            while(newend>newbeg)
                            {
                                if(s[newbeg]!=s[newend])
                                {
                                    return false;
                                }
                                ++newbeg;
                                --newend;
                            }
                            return true;
                        }
                        else
                        {
                            return false;
                        }
                    }
                    ++beg;
                    --end;
                }   
                return true;
            }
    };
    ```
* 代码2（答案思路）用时112ms：
    ```C++
    class Solution {
        public:
            bool sub(const string &s,int beg,int end)//使用引用降低时间
            {
                while(end>beg)
                {
                    if(s[end]!=s[beg])
                    {
                        return false;
                    }
                    ++beg;
                    --end;
                }
                return true;
            }
            bool validPalindrome(string s) {
                int beg = 0,end = s.size()-1;
                while(end>beg)
                {
                    if(s[beg] != s[end])
                    {
                        return sub(s,beg+1,end)||sub(s,beg,end-1);   
                    }
                    ++beg;
                    --end;
                }   
                return true;
            }
    };
    ```
