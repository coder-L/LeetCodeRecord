# 题目
给定一个链表，判断链表中是否有环。


* 思路：快慢指针
* 代码：
    ```C++
    /**
    * Definition for singly-linked list.
    * struct ListNode {
    *     int val;
    *     ListNode *next;
    *     ListNode(int x) : val(x), next(NULL) {}
    * };
    */
    class Solution {
    public:
        bool hasCycle(ListNode *head)
        {
            ListNode* quick = head;
            ListNode* slow = head;
            while(quick!=nullptr && slow!=nullptr)
            {
                if(quick->next==slow)
                {
                    return true;
                }
                if(quick->next == nullptr)
                {
                    return false;
                }
                quick = quick->next->next;
                slow = slow->next;
            }
            return false;
        }
    };
    ```
