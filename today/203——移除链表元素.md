# 题目
删除链表中等于给定值 val 的所有节点。

示例:

输入: 1->2->6->3->4->5->6, val = 6
输出: 1->2->3->4->5

* 代码：
    ```C++
    /**
    * Definition for singly-linked list.
    * struct ListNode {
    *     int val;
    *     ListNode *next;
    *     ListNode(int x) : val(x), next(NULL) {}
    * };
    */
    class Solution {
    public:
        ListNode* removeElements(ListNode* head, int val) {
            ListNode* result = new ListNode(0);
            result->next = head;
            ListNode* last = result;
            while(head!=nullptr)
            {
                if(head->val==val)
                {
                    ListNode* tmp = head;
                    head= head->next;
                    last->next = head;
                    delete tmp;
                }
                else
                {
                    head = head->next;
                    last = last->next;
                }
            }
            return result->next;
        }
    };
    ```
